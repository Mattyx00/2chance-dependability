        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\footnotesize, text width=12cm, inner sep=0.25cm},

            };

        \end{tikzpicture}

\begin{frame}
    \begin{center}
        \textcolor{FooterGreen}{\Large \textbf{Avoid SQL request in loop (GCI72)}}
        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\scriptsize, text width=12.5cm, inner sep=0.2cm},
            arrow/.style={->, thick, FooterGreen, >=Latex}
        ]
            
            \node[box] (problema) {
                \textbf{Problema} \\[1ex]
                Query SQL in cicli iterativi generano enorme overhead DBMS, bloccando i thread Tomcat e impedendo il basso consumo della CPU. Inefficienze rilevate:\\[0.5ex]
                $\bullet$ \textbf{Assenza Batch Processing}: singole \texttt{INSERT/UPDATE} in ciclo forzano transazioni e commit continui per ogni record, frammentando l'I/O su disco e saturando la rete.\\
                $\bullet$ \textbf{Problema N+1}: recuperare \textit{N} ordini e i loro prodotti costa \textit{1+N} query distinte (1 per ordini, \textit{N} per prodotti), moltiplicando roundtrip e attese I/O.
            };
            
            \node[box, below=0.3cm of problema] (soluzione) {
                \textbf{Soluzione} \\[1ex]
                $\bullet$ \textbf{Risoluzione Scritture}: introdotte operazioni \textbf{batch native JDBC} (\texttt{addBatch()}, \texttt{executeBatch()}). I comandi accodati vengono inviati in un singolo pacchetto di rete ed eseguiti in un'unica transazione atomica, abbattendo lock e costi di connessione.\\[0.5ex]
                $\bullet$ \textbf{Risoluzione N+1}: le \textit{1+N} query sono state ridotte a \textbf{1 query aggregata} (\texttt{LEFT JOIN}) in un solo roundtrip. L'assemblaggio senza duplicati è delegato alla RAM tramite \texttt{LinkedHashMap}, annullando del tutto gli sprechi di rete.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
    \end{center}
\end{frame}

\begin{frame}
    \begin{center}
        \textcolor{FooterGreen}{\Large \textbf{Do not call a function when declaring a for-type loop (GCI69)}}
        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\footnotesize, text width=12cm, inner sep=0.25cm},
            arrow/.style={->, thick, FooterGreen, >=Latex}
        ]
            
            \node[box] (problema) {
                \textbf{Problema} \\[1ex]
                Invocare funzioni \textbf{invarianti} (il cui risultato non cambia mai durante l'esecuzione) all'interno dei parametri di dichiarazione o di guardia di un ciclo forza la CPU a ricalcolare l'intero metodo ad ogni singola iterazione, causando un enorme e inutile dispendio di calcolo.
            };
            
            \node[box, below=0.6cm of problema] (soluzione) {
                \textbf{Soluzione} \\[1ex]
                I risultati di queste funzioni sono stati estratti e salvati in \textbf{variabili primitive locali} \textit{prima} dell'inizio del ciclo.\\[1ex]
                Ciò elimina alla radice migliaia di invocazioni ridondanti, fungendo da vero e proprio \textit{caching} preventivo e consentendo alla JVM di ottimizzare pienamente l'esecuzione del blocco iterativo.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
    \end{center}
\end{frame}

\begin{frame}
    \begin{center}
        \textcolor{FooterGreen}{\Large \textbf{Don't use the query SELECT * FROM (GCI74)}}
        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\footnotesize, text width=12cm, inner sep=0.25cm},
            arrow/.style={->, thick, FooterGreen, >=Latex}
        ]
            
            \node[box] (problema) {
                \textbf{Problema} \\[1ex]
                L'uso sistematico di \texttt{SELECT *} costringe il database a pesanti operazioni di I/O su disco per estrarre \textbf{tutti i campi}, inclusi quelli inutili o enormi (es. immagini in \textit{Base64}). Ciò genera traffico di rete sprecato e satura la JVM con massicce allocazioni di oggetti temporanei.
            };
            
            \node[box, below=0.6cm of problema] (soluzione) {
                \textbf{Soluzione} \\[1ex]
                Le query sono state riscritte esplicitando nella proiezione SQL \textbf{unicamente le colonne necessarie}.\\[1ex]
                Il taglio del \textit{payload} di rete alleggerisce la fase di mappatura ad oggetti (ORM), abbattendo le allocazioni nella memoria Heap e minimizzando l'intenso stress energetico causato dai cicli continui di \textbf{Garbage Collection}. L'allineamento dei risultati ai nomi simbolici delle colonne aumenta inoltre la resilienza strutturale del codice.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
    \end{center}
\end{frame}

\begin{frame}
    \begin{center}
        \textcolor{FooterGreen}{\Large \textbf{Fase 1: Isolamento Totale con Nuova JVM}}
        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\footnotesize, text width=12.5cm, inner sep=0.25cm},
            arrow/.style={->, thick, FooterGreen, >=Latex}
        ]
            
            \node[box] (problema) {
                \textbf{Il Problema delle Interferenze} \\[1ex]
                JMH normalmente esegue test multipli all'interno della stessa \textit{Java Virtual Machine} (JVM). Questo rischia di alterare severamente i consumi energetici, perché un test potrebbe subire il carico del \textit{Garbage Collector} innescato dal test precedente, sporcando i dati.
            };
            
            \node[box, below=0.5cm of problema] (soluzione) {
                \textbf{Avvio Isolato} \\[1ex]
                Lo script impone l'avvio di una JVM \textbf{completamente nuova e pulita} per ogni singolo metodo testato, incapsulandola all'interno del processo di EnergiBridge.\\[1ex]
                Ciò previene l'inquinamento della memoria \textit{Heap} e l'accavallamento dei thread, garantendo che i Joule registrati appartengano \textbf{esclusivamente} al frammento di codice in esame.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
    \end{center}

\begin{frame}
    \begin{center}
        \textcolor{FooterGreen}{\Large \textbf{Fase 2: Configurazione Microbenchmarking}}
        \vspace{0.4cm}
        
        \begin{tikzpicture}[
            box/.style={draw=FooterGreen, thick, rounded corners, text=white, align=left, fill=BgGreen!50!black, font=\footnotesize, text width=12.5cm, inner sep=0.25cm},
            arrow/.style={->, thick, FooterGreen, >=Latex}
        ]
            
            \node[box] (problema) {
                \textbf{L'Ottimizzazione JIT del Bytecode} \\[1ex]
                Il codice Java è pre-compilato in un formato intermedio (\textit{bytecode}). Durante l'avvio, il compilatore JIT (\textit{Just-In-Time}) della JVM lo traduce al volo in linguaggio macchina, applicando pesanti ottimizzazioni per l'hardware. Questo intenso processo di analisi e ottimizzazione del codice assorbe enormi quantità di energia che falsificherebbero i risultati.
            };
            
            \node[box, below=0.4cm of problema] (soluzione) {
                \textbf{Warmup e Misurazione a Singolo Thread} \\[1ex]
                Lo script imposta JMH per eseguire prima \textbf{5 iterazioni a vuoto} (\textit{warmup}). Queste iterazioni servono esclusivamente a concedere alla JVM il tempo e l'energia per profilare e ottimizzare a fondo il \textit{bytecode} in background.\\[1ex]
                Solo quando il codice è stato completamente ottimizzato dal JIT, lo script lancia le \textbf{5 iterazioni di misurazione reale} a singolo thread, registrando il vero costo energetico dell'algoritmo a regime.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
    \end{center}
\end{frame}

            \node[box] (problema) {
                \textbf{Rumore dei Sensori Hardware} \\[1ex]
                I sensori fisici integrati nel processore (registri \textit{RAPL}) possono generare rumore, cioé valori anomali matematicamente impossibili, come misurazioni con energia negativa, o picchi anomali di assorbimento in Watt che vanno oltre le capacità fisiche della CPU e dei singoli Core.
            };
            
            \node[box, below=0.4cm of problema] (soluzione) {
                \textbf{Filtraggio Matematico (TDP)} \\[1ex]
                Lo script applica un rigoroso filtro, scartando le letture in cui l'energia registrata ($\Delta E$) divisa per il tempo ($\Delta t$) supera il TDP (\textit{Thermal Design Power}) massimo fisicamente erogabile dalla CPU AMD usata per i test:\\[0.5ex]
                \hspace*{1cm} $0 \le \Delta E_{\text{CPU}} \le \textbf{55{,}0} \times \Delta t \quad \text{e} \quad 0 \le \Delta E_{\text{Core}} \le \textbf{45{,}0} \times \Delta t$\\[0.5ex]
                Ciò garantisce la totale \textbf{affidabilità empirica} dei misuramenti.
            };
            
            \draw[arrow] (problema) -- (soluzione);
            
        \end{tikzpicture}
